class Solution {
public:
    string mergeAlternately(string word1, string word2) {
        string res = "";
        
        // 交替合并公共长度
        int i = 0;
        while(i < word1.size() && i < word2.size()) {
            res += word1[i];
            res += word2[i];
            i++;
        }

        // 收尾
        while(i < word1.size()) res += word1[i++];
        while(i < word2.size()) res += word2[i++];

        return res;
    }
};

class Solution {
public:
    string gcdOfStrings(string str1, string str2) {
        int len = str1.size();
        if(str1.size() > str2.size()) swap(str1, str2);
        
        // 求最大，从大到小枚举长度
        for(int i = len; i >= 1; i--) {
            // 剪枝
            if(str1.size() % i || str2.size() % i) continue;
    
            string p = str1.substr(0, i);
            bool st = true;

            // 判断 p 能否除尽 str1，不能除尽就标记
            for(int j = 0; j + i - 1 < str1.size(); j += i)
                if(str1.substr(j, i) != p) {
                    st = 0;
                    break;
                }

            // 判断 p 能否除尽 str2
            for(int j = 0; j + i - 1 < str2.size(); j += i)
                if(str2.substr(j, i) != p) {
                    st = 0;
                    break;
                }

            // 都能除尽返回之大公因子串
            if(st) return p;
        }
        return "";
    }
};

class Solution {
public:
    vector<bool> kidsWithCandies(vector<int>& candies, int extraCandies) {
        // 求出当前最多糖果数
        int maxc = 0;
        for(int i = 0; i < candies.size(); i++)
            maxc = max(maxc, candies[i]);
        
        // 遍历
        vector<bool> res;
        for(int i = 0; i < candies.size(); i++) {
            if(candies[i] + extraCandies >= maxc)
                res.push_back(true);
            else
                res.push_back(false);
        }
        return res;
    }
};

class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        // 改变数组处理边界问题
        flowerbed.insert(flowerbed.begin(), 0); // 从 vector 容器头部插入
        flowerbed.push_back(0);
        
        // 枚举判断，满足情况就种一朵花
        for(int i = 1; i < flowerbed.size() - 1; i++) {
            if(flowerbed[i] == 0 && flowerbed[i - 1] == 0 && flowerbed[i + 1] == 0) {
                flowerbed[i] = 1;
                n--;
            }
        }

        return n <= 0;
    }
};

class Solution {
public:
    string reverseVowels(string s) {
        string ans = "";
        
        for(int i = 0; i < s.size(); i++)
            if(s[i] == 'a' || s[i] == 'A' || s[i] == 'e' || s[i] == 'E' || 
            s[i] == 'i' || s[i] == 'I' || s[i] == 'o' || s[i] == 'O' || 
            s[i] == 'u' || s[i] == 'U')
                ans += s[i];
        
        reverse(ans.begin(), ans.end());

        int j = 0;
        for(int i = 0; i < s.size(); i++) {
            if(s[i] == 'a' || s[i] == 'A' || s[i] == 'e' || s[i] == 'E' || 
            s[i] == 'i' || s[i] == 'I' || s[i] == 'o' || s[i] == 'O' || 
            s[i] == 'u' || s[i] == 'U') {
                s[i] = ans[j++];
            }
        }

        return s;
    }
};
