#### auto.cfg

```.cfg
//消除血迹
bind shift "+speed;r_cleardecals"   
//左中右跳投
alias +jumpthrow"+jump;-attack;-attack2";
alias -jumpthrow -jump;
bind MOUSE5 +jumpthrow;
//取消走动摆动
cl_bob_lower_amt 0
cl_bobamt_lat 0
cl_bobamt_vert 0
//取消枪口抖动
viewmodel_recoil 0
//显示网络信息
net_graph 1
//x轴 默认0.022 4:3 0.0165
m_yam 0.0165
//语音
bind "MOUSE4" "+voicerecord"
//鼠标灵敏度
sensitivity "1.25"
//准星
cl_crosshaircolor 1  //颜色
cl_crosshairsize 2.0  //长度
cl_crosshairthinkness 0.6  //粗细
cl_crosshairgap "-3.5"  //准星的间距，数值越大准星越扩散
cl_crosshairusealpha 0 //准星透明度
//cl_crosshairalpha 999  //透明度
cl_crosshair_drawoutline 1  //是否开启轮廓 
cl_crosshair_outlinethickness 1.0//轮廓参数
cl_crosshairdot 0  //中心点
cl_crosshairstyle 4  //（0默认、1默认静态、2经典、3经典动态、4经典静态）
//设置准星色温，三种参数包括blue蓝\red红\green绿色
cl_crosshaircolor_b "0"
cl_crosshaircolor_r "255"
cl_crosshaircolor_g "0" 
//投掷准星
bind "CAPSLOCK" "toggle cl_crosshairsize 2 99999"



//结尾
...
host_writeconfig;
//end


```

#### 启动项

-novid -nojoy -d3d9ex -tickrate 128 +exec auto.cfg
